### Launching from command line

{% highlight bash %}
$ flatpak list 
$ flatpak run org.darktable.Darktable 

# removing package
$ flatpak uninstall <application-ID>
$ flatpak uninstall com.spotify.Client

$ flatpak uninstall --unused     #  similar to apt autoremove 
{% endhighlight %}
